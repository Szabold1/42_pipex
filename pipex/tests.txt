
------------------------  /dev/urandom  ------------------------
< /dev/urandom cat | head -n1 > outfile
./pipex /dev/urandom "cat" "head -n1" outfile

------------------------  no infile  ------------------------
< noFile ls -l | sort -k7 > outfile
./pipex noFile "ls -l" "sort -k7" outfile

------------------------  no infile, invalid cmd1  ------------------------
< noFile lsss -l | sort -k7 > outfile
./pipex noFile "lsss -l" "sort -k7" outfile

------------------------  no infile, invalid last cmd  ------------------------
< noFile ls -l | sooort -k7 > outfile
./pipex noFile "ls -l" "sooort -k7" outfile

------------------------  invalid cmd1  ------------------------
< infile lsss -l | sort -k7 > outfile
./pipex infile "lsss -l" "sort -k7" outfile

------------------------  invalid last cmd  ------------------------
< infile ls -l | sooort -k7 > outfile
./pipex infile "ls -l" "sooort -k7" outfile

------------------------  invalid cmd1 and invalid last cmd  ------------------------
< infile lsss -l | sooort -k7 > outfile
./pipex infile "lsss -l" "sooort -k7" outfile

------------------------  no outfile  ------------------------
< infile ls -l | sort -k7 > newOutfile1
./pipex infile "ls -l" "sort -k7" newOutfile2

------------------------  empty env  ------------------------
env -i ./pipex infile "ls -l" "sort -k7" outfile

------------------------  no PATH  ------------------------
unset PATH
< infile ls -l | sort -k7 > outfile
./pipex infile "ls -l" "sort -k7" outfile

------------------------  full path to cmd  ------------------------
< infile /bin/ls -l | sort -k7 > outfile
./pipex infile "/bin/ls -l" "sort -k7" outfile
